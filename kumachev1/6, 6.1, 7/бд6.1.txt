6.1.1 – выберите все поля из таблицы Заказчики, в которых заказчиков обслуживали продавцы 1001, 1002 и 1007;

	select * from customers where snum IN (1001, 1002, 1007);


6.1.2 – выберите все поля из таблицы Заказы, в которых заказы оформляли заказчики 2001, 2004 и 2008;

	select * from orders where cnum in (2001, 2004, 2008)


6.1.3 – выберите из таблицы Продавцы все записи, у которых коэффициент продавца от 10% до 13%;

	select * from salespeople where comm between 0.10 and 0.13


6.1.4 – выберите из таблицы Продавцы все записи, где номер продавца от 1001 до 1005;

	select * from salespeople where snum between 1001 and 1005


6.1.5– выберите из таблицы Заказчики все записи где имя заказчика от буквы А до G;

	select * from customers where cname between 'A%' and 'G%'


 6.1.6– выберите из таблицы Заказчики все записи где заказчики проживают в городах от буквы А до L;

	select * from customers where city between 'A%' and 'L%'


6.1.7– выберите из таблицы Заказы все заказы, которые были оформлены с 10.03.1990 по 10.05.1990;

	select * from orders where odate between '1990-03-10' and '1990-05-10'
